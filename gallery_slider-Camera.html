Estrutura Local. Coloque em seu header:

<script src="Galeria/Slider/I/meedel/jquery-1.7.2.min.js">

<link rel="stylesheet" id="camera-css"  href="Galeria/Slider/I/meedel/camera_1/css/camera.css" type="text/css" media="all">
<script type="text/javascript" src="Galeria/Slider/I/meedel/camera_1/js/jquery.mobile.customized.min.js"></script>
<script type="text/javascript" src="Galeria/Slider/I/meedel/camera_1/js/jquery.easing.1.3.js"></script>
<script type="text/javascript" src="Galeria/Slider/I/meedel/camera_1/js/camera.min.js"></script>
<script type="text/javascript">
	jQuery(function(){		 
		jQuery('#camera_wrap_1').camera({
			thumbnails: true, 
			height: '400px',
			loader: 'pie', // bar,pie
			pieDiameter: 38,
			piePosition: "rightTop",               
			time: 1500, // ms (1500 = 1.5 seconds)
			pagination: true               
		});
	});
</script>
