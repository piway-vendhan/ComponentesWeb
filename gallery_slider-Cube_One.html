Cube One

Cube One utiliza-se a tecnologia Flash Player

Only

Source Code

Faça o Download dos arquivos, e inclua em sua aplicação na seguinte estrutura...

Galeria/
+-- Slider/
|   +-- I/
|   |   +-- cube_one/
|   |   |   +-- images/
|   |   |   |   +-- img_1.jpg
|   |   |   |   +-- img_2.jpg
|   |   |   |   +-- img_3.jpg
|   |   |   |   +-- img_4.jpg
|   |   |   +-- js/
|   |   |   |   +-- swfobject/
|   |   |   |   |   +-- swfobject.js
|   |   |   +-- config.xml
|   |   |   +-- cu3er.swf

Estrutura Local. Coloque em seu header:

<script type="text/javascript" src="Galeria/Slider/I/cube_one/js/swfobject/swfobject.js"></script>
Html 
<!-- Cube One -->
<script type="text/javascript">
	var flashvars = {};
	flashvars.xml = "Galeria/Slider/I/cube_one/config.xml";
	flashvars.font = "Galeria/Slider/I/cube_one/font.swf"; 
	var attributes = {};
	attributes.wmode = "transparent";
	attributes.id = "slider";
	swfobject.embedSWF( "Galeria/Slider/I/cube_one/cu3er.swf",
						"cu3er-container", 
						"960", 
						"316", "9", "Galeria/Slider/I/cube_one/expressInstall.swf", 
						flashvars, attributes);
						
</script>

<div class="header_text_bg">    
  <div id="cu3er-container"> 
	<a href="http://www.adobe.com/go/getflashplayer"> 
		<img src="http://www.adobe.com/images/shared/download_buttons/get_flash_player.gif" 
			 alt="Get Adobe Flash player" /> 
	</a> 
  </div>
</div>		
<!-- Fim Cube One -->
